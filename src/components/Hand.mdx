{/* Produces a single hand layout, inline or in table-style format */}
{/* usage: <Hand [inline]>cards</hands> */}
{/* Cards consists of 4 strings separated by space. */}
{/* A single dot can be used for no-cards in a suit */}

import { symbols } from "@helpers/ChangeSuits.mjs";

{(function () {
const hand = props.children;
if (!hand) return;

      const chunks = hand.split(' ')
        .map((el) => el == '.' ? '' : el);

      if (chunks.length != 4) return;

      if (props.inline) {
        return (
          <>
            <span class="hand inline">
              <span class="suit spade">S</span>{chunks[0]} <span class="suit heart">H</span>{chunks[1]} <span class="suit diamond">D</span>{chunks[2]} <span class="suit club">C</span>{chunks[3]}
            </span>
          </>
        );
      }

      return (
        <>
          <div class="hand">
            <div><span class="suit spade">S</span>{chunks[0]}</div>
            <div><span class="suit heart">H</span>{chunks[1]}</div>
            <div><span class="suit diamond">D</span>{chunks[2]}</div>
            <div><span class="suit club">C</span>{chunks[3]}</div>
          </div>
        </>
      );

    })()}
